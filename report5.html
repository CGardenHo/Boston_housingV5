def performance_metric2(y_true, y_predict):
    """"""
    y_mean = float(sum(y_true))/len(y_true)
    temp01 = [(each-y_mean)*(each-y_mean) for each in y_true]
    sstot = sum(temp01)
    
    temp02 = [(a-b)*(a-b) for a,b in zip(y_true, y_predict)]
    ssres = sum(temp02)
    
    score = 1 - float(ssres)/sstot

    return score

    #  [X,y]
    grid = grid.fit(X, y)
